## P0-1: eval() 安全隐患修复

### 问题定位
两处 eval() 使用：
1. **系统.py `数学计算`操作**（已有双重防护：正则白名单 + `{"__builtins__":{}}`）— 风险较低
2. **工具声明.json `计算器`工具**（无正则白名单，locals注入了abs/round/min/max/sum/pow/int/float/str/len/range/list/dict）— **风险较高**

### 方案：自研安全表达式解析器

```
公共区/内核/安全计算器.py  （新建，约150行）
```

**核心思路**：用Python标准库 `ast` 模块解析表达式为AST，只允许安全节点类型，自实现求值。

```python
import sqlite3, json, threading, pathlib

class 存储引擎类:
    """SQLite封装，零外部依赖（sqlite3是标准库），线程安全"""

    def __init__(self, 路径: str):
        self._锁 = threading.Lock()
        self._连接 = sqlite3.connect(路径, check_same_thread=False)
        self._连接.row_factory = sqlite3.Row
        self._初始化表()

    def _初始化表(self):
        # 记忆索引表
        self._执行("""
            CREATE TABLE IF NOT EXISTS 记忆索引 (
                名称 TEXT PRIMARY KEY,
                描述 TEXT,
                类型 TEXT,
                标签 TEXT,  -- JSON数组
                创建时间 TEXT,
                更新时间 TEXT,
                内容 TEXT   -- 全文搜索用
            )
        """)
        # 对话记录表
        self._执行("""
            CREATE TABLE IF NOT EXISTS 对话记录 (
                id TEXT,
                角色 TEXT,
                内容 TEXT,
                时间 TEXT,
                推理过程 TEXT,
                对话ID TEXT
            )
        """)
        # FTS5全文搜索索引
        self._执行("""
            CREATE VIRTUAL TABLE IF NOT EXISTS 对话搜索
            USING fts5(内容, 对话ID)
        """)

    def 查询对话(self, 关键词: str, limit=20):
        """全文搜索对话内容"""
        rows = self._查询(
            "SELECT 对话ID, snippet(对话搜索) as 片段 FROM 对话搜索 WHERE 内容 MATCH ? LIMIT ?",
            [关键词, limit]
        )
        return [dict(r) for r in rows]
```

### 文档分块器设计
```python
class 文档分块器:
    """零依赖文档分块"""

    def 分块(self, 文本: str, 块大小=512, 重叠=64) -> list:
        """按句子边界+固定长度分块"""
        句子 = re.split(r'[。！？\.\!\?\n]+', 文本)
        块 = []
        当前块 = ""
        for 句 in 句子:
            if len(当前块) + len(句) > 块大小 and 当前块:
                块.append(当前块.strip())
                当前块 = 当前块[-重叠:] + 句  # 保留重叠部分
            else:
                当前块 += 句
        if 当前块:
            块.append(当前块.strip())
        return 块
```

### BM25实现
```python
import math, re
from collections import Counter

class BM25索引:
    """纯标准库BM25实现"""

    def __init__(self, 文档列表: list):
        self._文档 = 文档列表
        self._分词 = [self._分词(文档) for 文档 in 文档列表]
        self._文档频率 = Counter()
        for 分词结果 in self._分词:
            for 词 in set(分词结果):
                self._文档频率[词] += 1
        self._平均长度 = sum(len(d) for d in self._分词) / len(self._分词)

    def 搜索(self, 查询: str, top_k=3) -> list:
        """返回最相关的top_k文档块"""
        查询词 = self._分词(查询)
        得分 = []
        for i, 文档分词 in enumerate(self._分词):
            s = 0
            for 词 in 查询词:
                if 词 not in self._文档频率:
                    continue
                # BM25公式
                idf = math.log((len(self._文档) - self._文档频率[词] + 0.5) /
                               (self._文档频率[词] + 0.5))
                tf = 文档分词.count(词)
                k1, b = 1.5, 0.75
                s += idf * (tf * (k1 + 1)) / (tf + k1 * (1 - b + b * len(文档分词) / self._平均长度))
            得分.append((i, s))
        得分.sort(key=lambda x: -x[1])
        return [(self._文档[i], s) for i, s in 得分[:top_k]]
```

### 中文分词
不用jieba（外部依赖），用正则提取：
```python
def _分词(self, 文本):
    # 2-gram滑窗分词（适合中文，无需词典）
    词 = re.findall(r'[\u4e00-\u9fff]+|[a-zA-Z]+|\d+', 文本)
    结果 = []
    for w in 词:
        if len(w) > 1:
            # 2-gram
            for i in range(len(w) - 1):
                结果.append(w[i:i+2])
        else:
            结果.append(w)
    return 结果
```

```python
class 剧本管理器:
    def 开始录制(self):
        self._录制中 = True
        self._当前剧本 = {"步骤": []}
        # 订阅操作注册中心的执行事件

    def _操作执行回调(self, 操作名, 参数, 结果):
        if self._录制中:
            self._当前剧本["步骤"].append({
                "操作": 操作名,
                "参数": 参数,
                "结果": 结果  # 供后续步骤引用
            })

    def 停止录制(self) -> dict:
        self._录制中 = False
        return self._当前剧本

    def 回放(self, 剧本: dict, 变量: dict = None):
        """逐步执行剧本，支持变量传递"""
        上下文 = 变量 or {}
        for 步骤 in 剧本["步骤"]:
            # 模板变量替换: ${步骤1.结果.字段}
            参数 = self._替换变量(步骤["参数"], 上下文)
            结果 = 操作注册中心.执行(步骤["操作"], 参数)
            上下文[f"步骤{步骤['序号']}"] = {"结果": 结果}
            if 步骤.get("等待完成"):
                # 同步等待
                pass
            else:
                # 异步执行
                pass

    def 保存剧本(self, 剧本: dict, 名称: str):
        路径 = pathlib.Path(剧本目录) / f"{名称}.json"
        with open(路径, 'w', encoding='utf-8') as f:
            json.dump(剧本, f, ensure_ascii=False, indent=2)

    def 定时触发(self, 剧本名: str):
        """供定时任务调度器调用"""
        剧本 = self.加载剧本(剧本名)
        self.回放(剧本)
```

```python
def _替换变量(self, obj, 上下文):
    """递归替换 ${变量名} 占位符"""
    if isinstance(obj, str):
        for 键, 值 in 上下文.items():
            obj = obj.replace(f"${{{键}}}", str(值))
        return obj
    elif isinstance(obj, dict):
        return {k: self._替换变量(v, 上下文) for k, v in obj.items()}
    elif isinstance(obj, list):
        return [self._替换变量(v, 上下文) for v in obj]
    return obj
```

```python
"""全局鼠标钩子 — 双击中键检测，零依赖（ctypes + Win32 API）"""
import ctypes
import threading
import time
import sys

# Win32 常量
WH_MOUSE_LL = 14
WM_MBUTTONDOWN = 0x0207
WM_MBUTTONUP = 0x020A
HC_ACTION = 0

# ctypes 结构体
class MSLLHOOKSTRUCT(ctypes.Structure):
    _fields_ = [
        ("pt", ctypes.c_long * 2),  # x, y
        ("mouseData", ctypes.c_ulong),
        ("flags", ctypes.c_ulong),
        ("time", ctypes.c_ulong),
        ("dwExtraInfo", ctypes.c_ulong),
    ]

class 全局呼出器:
    """监听全局鼠标中键双击，触发回调"""

    def __init__(self, 双击回调):
        self._回调 = 双击回调
        self._钩子句柄 = None
        self._线程 = None
        self._运行 = False
        self._上次中键时间 = 0
        self._双击间隔毫秒 = 400  # 双击判定阈值

    def 启动(self):
        if sys.platform != "win32":
            print("⚠️ 全局呼出器仅支持Windows")
            return
        self._运行 = True
        self._线程 = threading.Thread(target=self._消息循环, daemon=True)
        self._线程.start()

    def _消息循环(self):
        """设置低级鼠标钩子，接收全局鼠标事件"""
        # 定义钩子回调（LowLevelMouseProc）
        def 钩子回调(nCode, wParam, lParam):
            if nCode == HC_ACTION:
                if wParam in (WM_MBUTTONDOWN, WM_MBUTTONUP):
                    if wParam == WM_MBUTTONUP:  # 中键弹起时判定
                        当前时间 = int(time.time() * 1000)
                        if 当前时间 - self._上次中键时间 < self._双击间隔毫秒:
                            # 双击！
                            鼠标坐标 = self._获取鼠标坐标()
                            窗口标题 = self._获取前台窗口标题()
                            if self._回调:
                                self._回调(鼠标坐标, 窗口标题)
                            self._上次中键时间 = 0  # 重置，防三击
                        else:
                            self._上次中键时间 = 当前时间
            return ctypes.windll.user32.CallNextHookEx(
                self._钩子句柄, nCode, wParam, lParam
            )

        # 注册钩子
        HOOKPROC = ctypes.WINFUNCTYPE(
            ctypes.c_long,   # 返回值
            ctypes.c_int,    # nCode
            ctypes.c_uint,   # wParam
            ctypes.c_void_p  # lParam
        )
        self._钩子函数 = HOOKPROC(钩子回调)  # 防止GC回收
        self._钩子句柄 = ctypes.windll.user32.SetWindowsHookExW(
            WH_MOUSE_LL, self._钩子函数, None, 0
        )

        # 消息循环
        msg = ctypes.c_ulong()
        while self._运行:
            ctypes.windll.user32.GetMessageW(ctypes.byref(msg), None, 0, 0)
            ctypes.windll.user32.TranslateMessage(ctypes.byref(msg))
            ctypes.windll.user32.DispatchMessageW(ctypes.byref(msg))

    def _获取鼠标坐标(self):
        """获取当前鼠标屏幕坐标"""
        class POINT(ctypes.Structure):
            _fields_ = [("x", ctypes.c_long), ("y", ctypes.c_long)]
        pt = POINT()
        ctypes.windll.user32.GetCursorPos(ctypes.byref(pt))
        return (pt.x, pt.y)

    def _获取前台窗口标题(self):
        """获取当前前台窗口标题（告诉AI用户在干什么）"""
        hwnd = ctypes.windll.user32.GetForegroundWindow()
        length = ctypes.windll.user32.GetWindowTextLengthW(hwnd)
        buf = ctypes.create_unicode_buffer(length + 1)
        ctypes.windll.user32.GetWindowTextW(hwnd, buf, length + 1)
        return buf.value

    def 停止(self):
        self._运行 = False
        if self._钩子句柄:
            ctypes.windll.user32.UnhookWindowsHookEx(self._钩子句柄)
```

```python
        x = 鼠标坐标[0] - 窗口大小 // 2
        y = 鼠标坐标[1] - 窗口大小 // 2

        # 绘制扇区
        self._绘制轮盘()

        self.根窗口.bind("<Escape>", lambda e: self._关闭())
        # 点击外部关闭
        self.根窗口.bind("<FocusOut>", lambda e: self._关闭())

        self._淡入()

    def _绘制轮盘(self):
        """绘制扇区 + 中心圆"""
        self.扇区列表 = []
        扇区配置 = self.配置.get("扇区", [])
        n = len(扇区配置)
        if n == 0:
            return

        cx = self.中心[0]
        cy = self.中心[1]
        # 修正：Canvas坐标系是窗口左上角为原点
        cx = (self.半径 + 60)
        cy = (self.半径 + 60)

        for i, 扇区 in enumerate(扇区配置):
            # 每个扇区的角度范围
            角度起始 = -90 + 360 * i / n  # 从顶部开始顺时针
            角度结束 = -90 + 360 * (i + 1) / n
            颜色 = 扇区.get("颜色", "#2d2d44")
            图标 = 扇区.get("图标", "")
            名称 = 扇区.get("名称", "")
            说明 = 扇区.get("说明", "")

            # 绘制扇形
            arc_id = self.画布.create_arc(
                cx - self.半径, cy - self.半径,
                cx + self.半径, cy + self.半径,
                start=角度起始, extent=360 / n,
                fill=颜色, outline="#1a1a2e", width=2
            )

            # 计算图标位置（扇区中心）
            中角 = math.radians(角度起始 + (360 / n) / 2)
            图标r = self.半径 * 0.65
            ix = cx + 图标r * math.cos(中角)
            iy = cy + 图标r * math.sin(中角)

            图标id = self.画布.create_text(
                ix, iy, text=f"{图标}\n{名称}",
                fill="white", font=("Segoe UI Emoji", 14, "bold"),
                justify="center"
            )

            self.扇区列表.append({
                "arc_id": arc_id, "图标id": 图标id,
                "配置": 扇区, "默认色": 颜色,
                "角度起始": 角度起始, "角度结束": 角度结束
            })

        # 中心圆
        self.中心圆 = self.画布.create_oval(
            cx - self.中心圆半径, cy - self.中心圆半径,
            cx + self.中心圆半径, cy + self.中心圆半径,
            fill="#4d4d6e", outline="#5d5d7e", width=2
        )
        self.中心文字 = self.画布.create_text(
            cx, cy, text="⚡\n取消", fill="white",
            font=("Segoe UI Emoji", 10), justify="center"
        )

    def _鼠标移动(self, 事件):
        """hover检测：鼠标在哪个扇区"""
        cx = self.半径 + 60
        cy = self.半径 + 60
        dx = 事件.x - cx
        dy = 事件.y - cy
        距离 = math.sqrt(dx * dx + dy * dy)

        # 在中心圆内 → 高亮中心
        if 距离 < self.中心圆半径:
            if self.当前hover != -1:
                self._取消高亮(self.当前hover)
                self.当前hover = -1
                self.画布.itemconfig(self.中心圆, fill="#5d5d7e")
            return

        # 在轮盘范围内 → 计算角度
        if 距离 < self.半径:
            角度 = math.degrees(math.atan2(dy, dx))
            for i, 扇区 in enumerate(self.扇区列表):
                起始 = 扇区["角度起始"]
                结束 = 扇区["角度结束"]
                # 标准化角度
                if self._角度在范围内(角度, 起始, 结束):
                    if self.当前hover != i:
                        if self.当前hover >= 0:
                            self._取消高亮(self.当前hover)
                        self._高亮扇区(i)
                        self.当前hover = i
                    return

        # 在轮盘外
        if self.当前hover >= 0:
            self._取消高亮(self.当前hover)
            self.当前hover = -1

    def _角度在范围内(self, 角度, 起始, 结束):
        """检查角度是否在扇区范围内（处理跨0°情况）"""
        while 角度 < 起始:
            角度 += 360
        return 起始 <= 角度 < 结束

    def _高亮扇区(self, i):
        """扇区hover高亮：放大 + 发光色"""
        扇区 = self.扇区列表[i]
        颜色 = 扇区["配置"].get("颜色", "#4a9eff")
        # 亮色版本（简单提亮）
        亮色 = self._提亮颜色(颜色, 30)
        self.画布.itemconfig(扇区["arc_id"], fill=亮色, width=3)
        self.画布.itemconfig(扇区["图标id"], font=("Segoe UI Emoji", 16, "bold"))

    def _取消高亮(self, i):
        """取消扇区高亮"""
        扇区 = self.扇区列表[i]
        self.画布.itemconfig(扇区["arc_id"], fill=扇区["默认色"], width=2)
        self.画布.itemconfig(扇区["图标id"], font=("Segoe UI Emoji", 14, "bold"))

    def _提亮颜色(self, 颜色, 幅度):
        """将十六进制颜色提亮"""
        r = int(颜色[1:3], 16)
        g = int(颜色[3:5], 16)
        b = int(颜色[5:7], 16)
        r = min(255, r + 幅度)
        g = min(255, g + 幅度)
        b = min(255, b + 幅度)
        return f"#{r:02x}{g:02x}{b:02x}"

    def _点击(self, 事件):
        """点击扇区执行动作"""
        if self.当前hover == -1:
            # 点中心 → 取消
            self._关闭()
            return

        扇区 = self.扇区列表[self.当前hover]
        动作 = 扇区["配置"]["名称"]

        # 关闭轮盘，弹出回答区
        self._关闭轮盘保留窗口()
        self._执行动作(动作)

    def _执行动作(self, 动作):
        """根据动作名称执行，流式调用LLM"""
        # 获取选中文本（剪贴板或API）
        选中文本 = self._获取选中文本()

        # 根据动作构建消息
        if 动作 == "翻译":
            消息 = [{"role": "user", "content": f"翻译到中文，只输出译文：\n{选中文本}"}]
            提示词 = "你是翻译助手，只输出译文，不加解释。"
        elif 动作 == "识图":
            # 截图 → base64
            图片b64 = self._截图base64()
            消息 = [{"role": "user", "content": [
                {"type": "image_url", "image_url": {"url": f"data:image/png;base64,{图片b64}"}},
                {"type": "text", "text": "简洁描述图片内容"}
            ]}]
            提示词 = "你是图片分析助手，简洁描述图片内容。"
        elif 动作 == "问答":
            上下文 = f"[当前程序: {self.当前窗口标题}]"
            消息 = [{"role": "user", "content": f"{上下文}\n{选中文本}"}]
            提示词 = self.配置.get("系统提示词", "你是快速助手，简洁回答。")
        elif 动作 == "朗读":
            # 直接TTS，不调LLM
            if 选中文本:
                self.TTS回调(选中文本)
            self._关闭()
            return
        else:
            # 自定义动作
            消息 = [{"role": "user", "content": 选中文本}]
            提示词 = self.配置.get("系统提示词", "你是快速助手，简洁回答。")

        # 注入轻量记忆
        消息 = self._注入记忆(消息)

        # 展示回答区
        self._展示回答区()

        # 流式调用
        def 回调(片段):
            self.根窗口.after(0, lambda: self._追加文本(片段))
            # 流式TTS（可选：积累一段再朗读）

        threading.Thread(
            target=self._流式调用,
            args=(消息, 提示词, 回调),
            daemon=True
        ).start()

    def _注入记忆(self, 消息):
        """注入用户画像 + 最近5轮对话缓冲"""
        记忆配置 = self.配置.get("记忆", {})
        注入列表 = []

        # 用户画像（dict引用，零IO）
        if 记忆配置.get("注入用户画像", True):
            画像 = self.获取用户画像()
            if 画像:
                摘要 = f"[用户画像] 兴趣: {画像.get('兴趣关键词', {})} 偏好: {画像.get('学习到的偏好', {})}"
                注入列表.append({"role": "system", "content": 摘要})

        # 本地对话缓冲（内存list切片）
        if self.对话缓冲:
            注入列表.extend(self.对话缓冲[-self.缓冲上限:])

        return 注入列表 + 消息

    def _流式调用(self, 消息, 提示词, 回调):
        """调用模型直连器的流式接口"""
        结果 = self.模型直连器.发送消息流式(
            消息列表=消息,
            系统提示词=提示词,
            流式回调=回调
        )
        完整回复 = 结果.get("回复", "")

        # 追加到本地缓冲
        消息原文 = 消息[-1].get("content", "")
        if isinstance(消息原文, str):
            self.对话缓冲.append({"role": "user", "content": 消息原文})
        self.对话缓冲.append({"role": "assistant", "content": 完整回复})

        # TTS朗读完整回复
        if self.配置.get("自动朗读", False) and 完整回复:
            self.TTS回调(完整回复)

    def _获取选中文本(self):
        """获取当前选中文本（模拟Ctrl+C）"""
        import subprocess
        try:
            # 备份旧剪贴板
            result = subprocess.run(
                ["powershell", "-Command", "Get-Clipboard"],
                capture_output=True, text=True, timeout=1
            )
            旧内容 = result.stdout.strip()
            # 发送Ctrl+C
            ctypes.windll.user32.keybd_event(0x11, 0, 0, 0)  # Ctrl down
            ctypes.windll.user32.keybd_event(0x43, 0, 0, 0)  # C down
            ctypes.windll.user32.keybd_event(0x43, 0, 2, 0)  # C up
            ctypes.windll.user32.keybd_event(0x11, 0, 2, 0)  # Ctrl up
            time.sleep(0.1)
            # 读取新剪贴板
            result = subprocess.run(
                ["powershell", "-Command", "Get-Clipboard"],
                capture_output=True, text=True, timeout=1
            )
            新内容 = result.stdout.strip()
            if 新内容 and 新内容 != 旧内容:
                return 新内容
            return ""
        except Exception:
            return ""

    def _截图base64(self):
        """截取全屏，返回base64"""
        from PIL import ImageGrab
        import io, base64
        img = ImageGrab.grab()
        buf = io.BytesIO()
        img.save(buf, format="PNG")
        return base64.b64encode(buf.getvalue()).decode()

    def _展示回答区(self):
        """关闭轮盘，在原位置展示回答区"""
        # 清除Canvas上的轮盘
        self.画布.delete("all")
        # 绘制回答区
        self.根窗口.geometry(f"400x200+{self.中心[0]-200}+{self.中心[1]}")
        self.回答文本 = tk.Text(
            self.画布, bg="#1a1a2e", fg="#e0e0e0",
            font=("Consolas", 12), wrap="word",
            padx=15, pady=15, highlightthickness=0
        )
        self.回答文本.pack(fill="both", expand=True)
        # 点击外部关闭
        self.画布.bind("<Button-1>", lambda e: self._关闭())

    def _追加文本(self, 片段):
        """流式追加文本到回答区"""
        self.回答文本.insert("end", 片段)
        self.回答文本.see("end")

    def _淡入(self):
        """alpha从0淡入到目标透明度"""
        当前 = 0.0
        步长 = self.透明度 / (self.动画毫秒 / 16)  # ~60fps
        def 步进():
            nonlocal 当前
            当前 += 步长
            if 当前 >= self.透明度:
                self.根窗口.attributes("-alpha", self.透明度)
            else:
                self.根窗口.attributes("-alpha", 当前)
                self.根窗口.after(16, 步进)
        步进()

    def _关闭轮盘保留窗口(self):
        """关闭轮盘图形但保留窗口用于回答区"""
        pass  # _展示回答区 会清除Canvas重绘

    def _关闭(self, 事件=None):
        """关闭浮窗"""
        if self.根窗口:
            # 淡出
            当前 = self.透明度
            def 步进():
                nonlocal 当前
                当前 -= 0.1
                if 当前 <= 0:
                    self.根窗口.destroy()
                    self.根窗口 = None
                else:
                    try:
                        self.根窗口.attributes("-alpha", 当前)
                        self.根窗口.after(16, 步进)
                    except:
                        pass
            步进()

    def 停止(self):
        self._关闭()
```

| 零外部依赖 | ctypes+tkinter+PIL，全标准库（PIL已在系统中使用） |
