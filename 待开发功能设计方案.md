## P0-1: eval() 安全隐患修复

### 问题定位
两处 eval() 使用：
1. **系统.py `数学计算`操作**（已有双重防护：正则白名单 + `{"__builtins__":{}}`）— 风险较低
2. **工具声明.json `计算器`工具**（无正则白名单，locals注入了abs/round/min/max/sum/pow/int/float/str/len/range/list/dict）— **风险较高**

### 方案：自研安全表达式解析器

```
公共区/内核/安全计算器.py  （新建，约150行）
```

**核心思路**：用Python标准库 `ast` 模块解析表达式为AST，只允许安全节点类型，自实现求值。

```python
import sqlite3, json, threading, pathlib

class 存储引擎类:
    """SQLite封装，零外部依赖（sqlite3是标准库），线程安全"""

    def __init__(self, 路径: str):
        self._锁 = threading.Lock()
        self._连接 = sqlite3.connect(路径, check_same_thread=False)
        self._连接.row_factory = sqlite3.Row
        self._初始化表()

    def _初始化表(self):
        # 记忆索引表
        self._执行("""
            CREATE TABLE IF NOT EXISTS 记忆索引 (
                名称 TEXT PRIMARY KEY,
                描述 TEXT,
                类型 TEXT,
                标签 TEXT,  -- JSON数组
                创建时间 TEXT,
                更新时间 TEXT,
                内容 TEXT   -- 全文搜索用
            )
        """)
        # 对话记录表
        self._执行("""
            CREATE TABLE IF NOT EXISTS 对话记录 (
                id TEXT,
                角色 TEXT,
                内容 TEXT,
                时间 TEXT,
                推理过程 TEXT,
                对话ID TEXT
            )
        """)
        # FTS5全文搜索索引
        self._执行("""
            CREATE VIRTUAL TABLE IF NOT EXISTS 对话搜索
            USING fts5(内容, 对话ID)
        """)

    def 查询对话(self, 关键词: str, limit=20):
        """全文搜索对话内容"""
        rows = self._查询(
            "SELECT 对话ID, snippet(对话搜索) as 片段 FROM 对话搜索 WHERE 内容 MATCH ? LIMIT ?",
            [关键词, limit]
        )
        return [dict(r) for r in rows]
```

### 文档分块器设计
```python
class 文档分块器:
    """零依赖文档分块"""

    def 分块(self, 文本: str, 块大小=512, 重叠=64) -> list:
        """按句子边界+固定长度分块"""
        句子 = re.split(r'[。！？\.\!\?\n]+', 文本)
        块 = []
        当前块 = ""
        for 句 in 句子:
            if len(当前块) + len(句) > 块大小 and 当前块:
                块.append(当前块.strip())
                当前块 = 当前块[-重叠:] + 句  # 保留重叠部分
            else:
                当前块 += 句
        if 当前块:
            块.append(当前块.strip())
        return 块
```

### BM25实现
```python
import math, re
from collections import Counter

class BM25索引:
    """纯标准库BM25实现"""

    def __init__(self, 文档列表: list):
        self._文档 = 文档列表
        self._分词 = [self._分词(文档) for 文档 in 文档列表]
        self._文档频率 = Counter()
        for 分词结果 in self._分词:
            for 词 in set(分词结果):
                self._文档频率[词] += 1
        self._平均长度 = sum(len(d) for d in self._分词) / len(self._分词)

    def 搜索(self, 查询: str, top_k=3) -> list:
        """返回最相关的top_k文档块"""
        查询词 = self._分词(查询)
        得分 = []
        for i, 文档分词 in enumerate(self._分词):
            s = 0
            for 词 in 查询词:
                if 词 not in self._文档频率:
                    continue
                # BM25公式
                idf = math.log((len(self._文档) - self._文档频率[词] + 0.5) /
                               (self._文档频率[词] + 0.5))
                tf = 文档分词.count(词)
                k1, b = 1.5, 0.75
                s += idf * (tf * (k1 + 1)) / (tf + k1 * (1 - b + b * len(文档分词) / self._平均长度))
            得分.append((i, s))
        得分.sort(key=lambda x: -x[1])
        return [(self._文档[i], s) for i, s in 得分[:top_k]]
```

### 中文分词
不用jieba（外部依赖），用正则提取：
```python
def _分词(self, 文本):
    # 2-gram滑窗分词（适合中文，无需词典）
    词 = re.findall(r'[\u4e00-\u9fff]+|[a-zA-Z]+|\d+', 文本)
    结果 = []
    for w in 词:
        if len(w) > 1:
            # 2-gram
            for i in range(len(w) - 1):
                结果.append(w[i:i+2])
        else:
            结果.append(w)
    return 结果
```

class 剧本管理器:
    def 开始录制(self):
        self._录制中 = True
        self._当前剧本 = {"步骤": []}
        # 订阅操作注册中心的执行事件

    def _操作执行回调(self, 操作名, 参数, 结果):
        if self._录制中:
            self._当前剧本["步骤"].append({
                "操作": 操作名,
                "参数": 参数,
                "结果": 结果  # 供后续步骤引用
            })

    def 停止录制(self) -> dict:
        self._录制中 = False
        return self._当前剧本

    def 回放(self, 剧本: dict, 变量: dict = None):
        """逐步执行剧本，支持变量传递"""
        上下文 = 变量 or {}
        for 步骤 in 剧本["步骤"]:
            # 模板变量替换: ${步骤1.结果.字段}
            参数 = self._替换变量(步骤["参数"], 上下文)
            结果 = 操作注册中心.执行(步骤["操作"], 参数)
            上下文[f"步骤{步骤['序号']}"] = {"结果": 结果}
            if 步骤.get("等待完成"):
                # 同步等待
                pass
            else:
                # 异步执行
                pass

    def 保存剧本(self, 剧本: dict, 名称: str):
        路径 = pathlib.Path(剧本目录) / f"{名称}.json"
        with open(路径, 'w', encoding='utf-8') as f:
            json.dump(剧本, f, ensure_ascii=False, indent=2)

    def 定时触发(self, 剧本名: str):
        """供定时任务调度器调用"""
        剧本 = self.加载剧本(剧本名)
        self.回放(剧本)

def _替换变量(self, obj, 上下文):
    """递归替换 ${变量名} 占位符"""
    if isinstance(obj, str):
        for 键, 值 in 上下文.items():
            obj = obj.replace(f"${{{键}}}", str(值))
        return obj
    elif isinstance(obj, dict):
        return {k: self._替换变量(v, 上下文) for k, v in obj.items()}
    elif isinstance(obj, list):
        return [self._替换变量(v, 上下文) for v in obj]
    return obj
```

                lines.append(f"## 🧑 用户\n\n{内容}\n")
            elif 角色 == "assistant":
                lines.append(f"## 🤖 助手\n\n{内容}\n")

            if 包含推理 and msg.get("推理过程"):
                lines.append(f"<details><summary>推理过程</summary>\n\n{msg['推理过程']}\n\n</details>\n")

            lines.append("---\n\n")

        return "\n".join(lines)
```

### 现有基础

### 方案：LLM生成工具声明JSON + 验证 + 热加载

**新增操作**：
```python
class 创建工具操作(操作基类):
    """AI根据用户描述自动生成工具声明JSON"""

    def 执行(self, 参数):
        描述 = 参数.get("描述", "")
        示例 = 参数.get("示例", "")

        # 1. 构建提示词，让LLM生成工具声明JSON
        提示词 = f"""
请根据以下描述生成一个工具声明JSON对象：
描述：{描述}
示例：{示例}

输出格式（严格遵守）：
{{
  "名称": "工具中文名",
  "描述": "工具说明",
  "类型": "http" 或 "python",
  "配置": {{...}},
  "参数": [{{"名称": "参数名", "类型": "string", "必填": true, "说明": "..."}}],
  "参数映射": {{"参数名": "参数名"}}
}}
"""
        # 2. 调用LLM生成
        结果 = 模型直连器.发送消息([{"role":"user","content":提示词}])
        声明JSON = self._提取JSON(结果)

        # 3. 验证JSON结构
        验证结果 = self._验证工具声明(声明JSON)
        if not 验证结果["有效"]:
            return 操作结果.失败(f"生成的工具声明无效: {验证结果['错误']}")

        # 4. 写入工具声明.json
        工具声明 = 配置加载器.读取配置("工具声明")
        工具声明["工具列表"].append(声明JSON)
        配置加载器.保存配置("工具声明", 工具声明)

        # 5. 热加载新工具
        动态工具加载器.重新加载()
        return 操作结果.成功(f"已创建工具: {声明JSON['名称']}")
```

**验证规则**：
- 名称非空且为中文
- 类型为 http 或 python
- http类型必须有URL配置
- python类型必须有脚本配置
- 参数列表结构正确
